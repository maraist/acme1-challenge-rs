Video ReCompose API Documentation
Abstract

This web-service initiates a video transcode pull-transcode-push pipeline stage.

It accepts a tripplet of web-URIs:

  • Two of which are sources which will be accessed via an HTTP-GET.
  • One of which will have the results pushed to via an HTTP-POST.

The service can be tested by self-routing; A sample video clip is available.

API
POST /recompose
Transcode a video asynchronously, returning a polling URL or error
Request Content Type: application/json
Request Payload: 
    type Request = {
        video_url: string; // URI that accepts GET, containing a source video
        rules_url: string; // URI that accepts GET, containing processing rules
        dest_url:  string; // URI that accepts POST, receives transcoded video file
    };
Sample Request 
    {
        "video_url": "https://www.example.com/input/video.rvid",
        "rules_url": "https://www.example.com/input/rules.json",
        "dest_url": "https://www.example.com/output/video.rvid",
    }
Response Content Type: application/json
Response Payload: 
    type ErrorItem = {
        status: string?; // A stringified HTTP status code
        title:  string?; // Short human display value for the reason of the error
        detail: string?; // Potentially longer description of the cause of the error.
        source: string?; // A link to this documentation describing the error
    };
    type DataItem = {
        /* If the status is RUNNING, this contains a URL that
         * can be GET-fetched to see the current status.
         */
        poll_url: string;
    };
    type Response = {
        status: "COMPLETED" | "RUNNING" | "ERROR";
        errors: Array<ErrorItem>?; // Only present when status === "ERROR"
        data:   DataItem?:         // Only present when status === "RUNNING"
    };
Sample Request 
    {
       "status": "RUNNING",
       "data": {
                    "poll_url": "http://recompose.api.recall.ai/recompose/poll?id=12345"
               }
    }

    {
       "status": "ERROR",
       "errors":
         [
           {
             "status": "422",
             "title": "required rules_url field is missing",
             "source": "http://recompose.api.recall.ai/static/index.html#rules_url_missing"
           }
         ]
    }

    {
       "status": "COMPLETED",
    }

status=r.get("status","ERROR")
if status == "ERROR":
  if "errors" in r:
      for err in r.errors:
          print(f"Error: {err.title}")
  return False
if status == "SUCCESS":

  }
  let encoded_message = JSON.stringify(msg)

  let options = {
                   method: 'POST',
                   body: encoded_message
                }
  let response = await fetch(server_url, options)
  const sleep = (delay)=> new Promise((res)=> setTimeout(res,delay))
  while (response.status === "RUNNING") {
       await sleep(5000);
       response = await fetch(response.data.poll_url)
  }
  if (response.status === "COMPLETED") {
    return true
  } else if (response.status === "ERROR") {

        console.log(`Could not process job: ${err.title}`)
    }
    return false
  }
